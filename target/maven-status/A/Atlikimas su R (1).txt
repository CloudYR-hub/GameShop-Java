# SPRENDIMAS

# Suvedame duomenis
x <- c(1, 1, 1, 4, 4, 5, 5, 4, 4, 5, 7, 6, 7, 6, 6, 10, 9)
print(x)

# Sudarome dažnių skirstinį
dazniu_skirstinys <- table(x)
print(dazniu_skirstinys)

# Pavaizduojame dažnių skirstinį grafiškai
plot(dazniu_skirstinys, type = "b",
     main = "Dažnių skirstinys", 
     ylab = "Dažniai",
     col = "red",
     pch = 21,
     bg = 2)

# Sudarome santykinių dažnių skirstinį
sant_dazniu_skirstinys <- table(x) / length(x)
print(sant_dazniu_skirstinys)

# Pavaizduojame santykinių dažnių skirstinį grafiškai
plot(sant_dazniu_skirstinys, type = "b",
     main = "Santykinių dažnių skirstinys", 
     ylab = "Santykiniai dažniai",
     col = "red",
     pch = 21,
     bg = 2)

# Imties variacinė eilutė
print(sort(x))

# Imties tūris
print(length(x))

# Mažiausioji imties reikšmė
print(min(x))

# Didžiausioji imties reikšmė
print(max(x))

# Imties mažiausioji ir didžiausioji reikšmės
print(range(x))

# Imties plotis
print(diff(range(x)))

# Imties moda
# PASTABA! Reikia instaliuoti paketą "modeest" (jei dar neinstaliuotas).
# install.packages("modeest")
library(modeest)
moda <- mfv(x)
print(moda)

# Imties mediana
print(median(x))

# Apskaičiuojame pirmąjį, antrąjį ir trečiąjį kvartilius
print(quantile(x, c(0.25, 0.5, 0.75), type = 2))

# Apskaičiuojame kvartilinį plotį
KP <- IQR(x)
print(KP)

# Suvestinė - minimali reikšmė, pirmasis kvartilis, mediana, vidurkis, trečiasis kvartilis, maksimali reikšmė
print(summary(x, quantile.type = 2))

# Apskaičiuojame 0.17-kvantilį
print(quantile(x, 0.17, type = 2))

# Apskaičiuojame 0.2-kvantilį, 0.47-kvantilį, 0.73-kvantilį
print(quantile(x, c(0.2, 0.47, 0.73), type = 2))

# Penkiaskaitė suvestinė - minimali reikšmė, pirmasis kvartilis, mediana, trečiasis kvartilis, maksimali reikšmė
print(fivenum(x))

# Stačiakampė diagrama
boxplot(x, horizontal = TRUE, main = "Stačiakampė diagrama")

# Empirinis vidurkis
print(mean(x))

# Empirinė pataisytoji dispersija
disp_pat <- var(x)
print(disp_pat)

# Empirinė dispersija
disp <- function(x) { var(x) * (length(x) - 1) / length(x) }
print(disp(x))

# Empirinis pataisytasis standartinis nuokrypis
standartas_pat <- sd(x)
print(standartas_pat)

# Empirinis standartinis nuokrypis
standartas <- sqrt(disp(x))
print(standartas)
