# TAŠKINIAI ĮVERTINIAI

## 1. Užduotis
# Svorio X taškinis įvertinys (kg) turi normalųjį skirstinį: X ~ N(μ, σ²)
# Duomenys: 9.8, 9.9, 10.0, 10.3. Raskite μ ir σ² taškinių momentų bei didžiausio tikėtinumo metodais.

X <- c(9.8, 9.9, 10.0, 10.3)

# Momentų metodas:
m <- mean(X)  # Vidurkis (μ)
n <- length(X)  # Imties dydis
disp <- ((n - 1) / n) * var(X)  # Dispersija (σ²)

# Didžiausio tikėtinumo metodas:
install.packages("MASS")
library(MASS)
iverciai <- fitdistr(X, "normal")  # Didžiausio tikėtinumo įverčiai

## 2. Užduotis
# Prietaiso veikimo trukmė (savaitėmis) pasiskirsčiusi eksponentiniu dėsniu: X ~ Exp(λ)
# Duomenys: 1236, 335, 119, 76, 454. Raskite λ taškinius įverčius.

X <- c(1236, 335, 119, 76, 454)

# Momentų metodas:
m <- mean(X)  # Vidurkis
lambda <- 1 / m  # Parametras λ

# Didžiausio tikėtinumo metodas:
fitdistr(X, "exponential")  # Didžiausio tikėtinumo įverčiai

## 3. Užduotis
# Skrydžių trukmė (valandomis) yra tolygiai pasiskirsčiusi: X ~ U(a, b)
# Duomenys: 8.00, 8.38, 9.11, 9.45, 8.23, 8.88. Raskite a ir b.

X <- c(8.00, 8.38, 9.11, 9.45, 8.23, 8.88)

# Parametrai:
a <- min(X)  # Mažiausia reikšmė (a)
b <- max(X)  # Didžiausia reikšmė (b)

## 4. Užduotis
# Atliekami keturi nepriklausomi matavimai, rezultatai: 2235, 2244, 2256, 2265.
# Matavimo vidutinė paklaida: 40 m. Nustatyti pasikliovimo intervalą (95% pasikliovimo lygmuo).

X <- c(2235, 2244, 2256, 2265)

# Pirmasis būdas (naudojant teorines formules):
n <- length(X)  # Imties dydis
Xvid <- mean(X)  # Vidurkis
sigma <- 40  # Paklaida
alfa <- 0.05  # Reikšmingumo lygmuo
p <- 1 - alfa / 2  # Tikimybė
zp <- qnorm(p)  # Kvantilis
PI <- c(Xvid - sigma * zp / sqrt(n), Xvid + sigma * zp / sqrt(n))  # Pasikliovimo intervalas

# Antrasis būdas (naudojant DescTools paketą):
install.packages("DescTools")
library(DescTools)
MeanCI(X, sd = sigma, conf.level = 1 - alfa)

## 5. Užduotis
# Iš 400 apklaustų žmonių (25–45 m.), 144 nuomojasi būstą.
# Nustatyti pasikliovimo intervalą su 95% pasikliovimo lygmeniu.

k <- 144  # Nuomojasi būstą
n <- 400  # Apklaustųjų skaičius
p <- k / n  # Tikimybės taškinis įvertis
alfa <- 0.05  # Reikšmingumo lygmuo
zp <- qnorm(1 - alfa / 2)  # Kvantilis

# Pasikliovimo intervalas (naudojant teorines formules):
PI <- c(round(p - zp * sqrt(p * (1 - p) / n), 3), round(p + zp * sqrt(p * (1 - p) / n), 3))

# Antrasis būdas (naudojant DescTools paketą):
install.packages("DescTools")
library(DescTools)
BinomCI(k, n, conf.level = 1 - alfa, method = "wald")

# Suderinamumo hipotezės apie normalųjį skirstinį

# Duomenys iš failo "Suderinamumo_hipotezes_imtys.csv"
duom <- read.csv("Suderinamumo_hipotezes_imtys.csv")
x <- duom$normalusis

n <- length(x) # 60

min_val <- min(x)
max_val <- max(x)
k <- ceiling(1 + 3.322 * log10(n)) # Skaičius intervalų

# Nustatome intervalų pradžią ir pabaigą
a_pirm <- floor(min_val)
a_pask <- ceiling(max_val)

# Sukuriame intervalus
bin <- seq(a_pirm, a_pask, length.out = k + 1)
print(bin)

# Sudarome intervalinį dažnių histograma
histograma <- hist(x, breaks = bin, 
                   plot = TRUE,
                   right = FALSE,
                   probability = TRUE, 
                   main = "Histograma",
                   ylab = 'Tankis')

# Pridėkite tankio kreivę
lines(density(x), col = "red", lwd = 3)

# Taškinė skirstinio lentelė
dažniai <- table(cut(x, breaks = bin, right = FALSE))
print(dažniai)

# Vertiname normalaus skirstinio parametrus
m <- mean(x)
sigma <- sd(x)

# Sukuriame intervalų ribas su -Inf ir Inf
binl <- c(-Inf, bin[-1], Inf)

# Skaičiuojame teorines tikimybes pagal normalųjį skirstinį
p <- pnorm(binl[-1], mean = m, sd = sigma) - pnorm(binl[-length(binl)], mean = m, sd = sigma)
print(p)

# Tikėtini dažniai
E <- n * p
print(E)

# Apskaičiuojame Chi-kvadrato statistiką
chi_kv_statistika <- sum((dažniai - E)^2 / E)
print(chi_kv_statistika)

# Apskaičiuojame Chi-kvadrato kvantilį
alpha <- 0.05
Chi_kv_kvantilis <- qchisq(1 - alpha, df = k - 1 - 1)
print(Chi_kv_kvantilis)

# Išvada: 2.131408 < 9.487729, todėl hipotezė neatsisakoma.
# Skirstinys atitinka normalųjį modelį.

# Antrasis būdas naudojant chisq.test()
test <- chisq.test(dažniai, p = E / sum(E))

# Koreguojame laisvės laipsnius
test$parameter <- k - 2
test$p.value <- pchisq(test$statistika, df = test$parameter, lower.tail = FALSE)
print(test)

# Išvada: p-value = 0.7116 > 0.05, todėl hipotezė neatsisakoma.

# Stebint atsitiktinį dydį X, buvo gauta imtis, įrašyta faile "Suderinamumo_hipotezes_imtys.csv".
# Patikrinkite, ar ši imtis pasiskirsto pagal tolygųjį skirstinį X ~ U(0, 2).

# Nuskaitome duomenis iš failo "Suderinamumo_hipotezes_imtys.csv".
# Kintamajam x priskiriame stulpelį "tolygusis", kuriame yra tiriami duomenys.
duom <- read.csv("Suderinamumo_hipotezes_imtys.csv")
x <- duom$tolygusis
x

# Skaičiuojame imties dydį n ir apibrėžiame intervalo ribas a ir b:
n <- length(x)
n
a <- 0
b <- 2

# Surikiuojame duomenis didėjimo tvarka:
x <- sort(x)
x

# Skaičiuojame teorines pasiskirstymo funkcijos reikšmes:
FoX <- punif(x, a, b)
FoX

# Skaičiuojame empirinės pasiskirstymo funkcijos reikšmes FnX (i/n) ir Fn1X ((i-1)/n):
FnX <- seq(1:n) / n
Fn1X <- (seq(1:n) - 1) / n

# Apskaičiuojame skirtumus su teorine funkcija:
Dp <- FnX - FoX  # Teorinis minus faktinis (aukštesnė riba)
Dm <- FoX - Fn1X # Faktinis minus teorinis (žemesnė riba)

# Sudarome lentelę su tarpiniais skaičiavimais:
Lentele <- cbind(x, FnX, Fn1X, FoX, Dp, Dm)
Lentele

# Iš lentelės išrenkame reikšmes Dp (aukščiausias skirtumas) ir Dm (žemiausias skirtumas):
DP <- max(Dp)
DP
DM <- max(Dm)
DM

# Skaičiuojame Kolmogorovo statistikos D_n reikšmę:
Dn <- max(DP, DM)
Dn

# Apibrėžiame reikšmingumo lygmenį alpha:
alpha <- 0.05

# Skaičiuojame Kolmogorovo skirstinio kritinę reikšmę k:
k <- sqrt(-log(alpha / 2) / (2 * n)) - 1 / (6 * n)
k

# Atsakymas: jei Dn < k, tai hipotezė neprištarauja duomenims.

# Antrasis būdas (naudojant R funkciją ks.test()):
# Funkcija ks.test() priima argumentus: imtis x, teorinės pasiskirstymo funkcijos pavadinimą, ir intervalo ribas.
ks.test(x, y = "punif", a, b)

# Atsakymas: Kadangi p-reikšmė didesnė už reikšmingumo lygmenį (pvz., 0.799 > 0.05),
# hipotezė, kad atsitiktinis dydis pasiskirstęs pagal tolydųjį skirstinį, neprištarauja imties duomenims.
